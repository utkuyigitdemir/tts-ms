# INSTALLATION (must be done manually - pip install won't work)

#
# 1. Clone repository:
#    git clone --recursive https://github.com/FunAudioLLM/CosyVoice.git
#    cd CosyVoice
#    git submodule update --init --recursive
#
# 2. Create conda environment (Python 3.10 required):
#    conda create -n cosyvoice python=3.10 -y
#    conda activate cosyvoice
#
# 3. Install system dependencies:
#    # Ubuntu/Debian:
#    sudo apt-get install sox libsox-dev
#    # CentOS/RHEL:
#    sudo yum install sox sox-devel
#
# 4. Install PyTorch (match your CUDA version):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/cu121
#
# 5. Install CosyVoice:
#    pip install -r requirements.txt
#
# 6. Download models:
#    # Models are downloaded automatically on first use, or manually:
